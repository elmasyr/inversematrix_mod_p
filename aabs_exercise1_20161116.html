from numpy import *
import numpy as np

s = 30 # start time in seconds
e = 45 # end time in seconds

df_temp=df[(df.sec>=s) & (df.sec<e)]
data = np.array(df_temp.signal)
lmax = ((diff(sign(diff(data))) < 0)).nonzero()[0] + 1 # local max

print len(lmax), 'peaks in', (e-s), 'seconds'
df_temp.plot(x=['sec'],y= ['signal'], figsize=(12, 6));

32 peaks in 15 seconds
